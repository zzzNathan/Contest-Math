\begin{problem}{HMMT Nov 2025 Teams --- P5}
Kelvin the frog is in the bottom-left cell of a $6 \times 6$ grid, and he
wants to reach the top-right cell.
He can take steps either up one cell or right one cell. However, there is a
raccoon in one of the 36 cells
uniformly at random, and Kelvin’s path must avoid this raccoon. Compute the
expected number of
distinct paths Kelvin can take to reach the top-right cell.
\vspace{0.2cm}

(If the raccoon is in either the bottom-left or top-right cell, then there
are 0 such paths.)
\end{problem}

\begin{problem}{HMMT Nov 2025 Teams --- P6}
Let $P$ be a point inside triangle $\triangle ABC$ such that $BP = PC$
and $\angle ABP + \angle ACP = 90^{\circ}$. Given
that $AB = 12, AC = 16$, and $AP = 11$, compute the area of the concave
quadrilateral $ABPC$.
\end{problem}

\begin{problem}{HMMT Nov 2025 Teams --- P9}
Let $a,b$ and $c$ be positive real numbers such that
\[
\begin{aligned}
\sqrt{a} + \sqrt{b} + \sqrt{c} &= 7\\
\sqrt{a+1} + \sqrt{b+1} + \sqrt{c+1} &= 8\\
(\sqrt{a+1} + \sqrt{a})(\sqrt{b+1} + \sqrt{b})(\sqrt{c+1} + \sqrt{c}) &= 60\\
\end{aligned}
\]
Compute $a+b+c$.
\end{problem}

\begin{problem}{HMMT Nov 2025 Guts --- P7}
Point $X$ lies on diagonal $AC$ of rectangle $ABCD$ such that
$AX = 11, CX = 1$, and triangle $BXD$
has area 18. Given that $BX < DX$, compute $BX$.
\end{problem}

\begin{problem}{HMMT Nov 2024 Teams --- P6}
There are 5 people who start with 1, 2, 3, 4, and 5 cookies, respectively.
Every minute, two
different people are chosen uniformly at random. If they have $a$ and $b$ cookies and $a\neq b$, the person
with more cookies eats $|a - b|$ of their own cookies. If $a = b$, the
minute still passes with nothing
happening. \vspace{0.2cm}

Compute the expected number of minutes until all 5 people have an equal number of cookies.
\end{problem}

\begin{problem}{BMO 1 2018 P4/6}
Let $\Gamma$ be a semicircle with diameter $AB$. The point $C$ lies on the
diameter $AB$ and points $E$ and $D$ lie on the arc $BA$, with $E$ between
$B$ and $D$. Let the tangents to $\Gamma$ at $D$ and $E$ meet at $F$.
Suppose that $\angle ACD = \angle ECB$.\vspace{0.2cm}

Prove that $\angle EFD = \angle ACD + \angle ECB$.
\end{problem}
