\begin{problem}{Australian Math Olympiad 2018 --- P5/8}
The sequence $a_1, a_2, a_3, \ldots$ is defined by $a_1 = 1$, and for
$n \geq 2$,

$$
a_n = (a_1 + a_2 + \cdots + a_{n-1}) \times n.
$$

Prove that $a_{2018}$ is divisible by $2018^2$.
\end{problem}

\begin{solution}{Solution}
Let $n \geq 3$.
Since $a_{n-1} = (a_1 + a_2 + \cdots + a_{n-2})(n-1)$ we deduce that

$$
a_n = a_{n-1} \frac{n}{n-1} + na_{n-1} = a_{n-1} \frac{n^2}{n-1}
$$

but since $a_{n-1}$ has a factor of $n-1$, we must have $n^2 \mid a_n$, and
thus $2018^2 \mid a_{2018}$ as desired.
\end{solution}
