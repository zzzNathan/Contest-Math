\textit{Selected solves from Evan Chen's Online Math Olympiad 2013.}

\begin{problem}{Problem 12}
Let $a_n$ denote the remainder when $(n+1)^3$ is divided by $n^3$; in
particular $a_1 = 0$. Compute the remainder when
$a_1 + a_2 + \cdots + a_{2013}$ is divided by 1000.
\end{problem}

\begin{solution}{Solution}
The largest multiple of $n^3$ that is less than $(n+1)^3$
is
$$
\floor{\frac{(n+1)^3}{n^3}}n^3 = n^3 + \floor{\frac{3n^2 + 3n + 1}{n^3}}
n^3.
$$
Thus
$$
a_n = (n+1)^3 - \left[ n^3 + \floor{\frac{3n^2 + 3n + 1}{n^3}} n^3 \right].
$$
It is easy to see that when $n \geq 4$ the fraction being floored is $< 1$
and so that whole term goes to 0. Hence from $n = 4, 5, \ldots, 2013$
we can telescope the sum like so
$$
\sum_{n=4}^{2013} a_n = \sum_{n = 4}^{2013} (n+1)^3 - n^3 = 2014^3 - 4^3.
$$
Then it is easy to just compute $a_1 = 0, a_2 = 3, a_3 = 10$.
So the problem reduces to evaluating

\[
\begin{aligned}
2014^3 - 4^3 + 3 + 10 &\equiv 14^3 - 4^3 + 3 + 10 \ (\text{mod } 1000)\\
&\equiv 2693 \equiv 693 \ (\text{mod } 1000)
\end{aligned}
\]

So the answer is $\boxed{693}$.
\end{solution}
